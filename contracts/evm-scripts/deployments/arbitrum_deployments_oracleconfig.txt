# Oracle ownership handoff (Arbitrum mainnet example):
# With NETWORK=arbitrum, oracle addresses load from deployments/oracle_deployments/arbitrumone/addresses.json
# unless you set overrides. ORACLE_PROXY_ADMIN_ADDRESS must be the DefaultProxyAdmin contract from that book — never GlobalConfig.
MODE=LIVE NETWORK=arbitrum BROADCAST=true just admin-oracle-transfer-to-globalconfig

MODE=LIVE NETWORK=arbitrum ACCOUNT_TO_PERMIT="$GLOBALCONFIG_ADDRESS" TARGET_ADDRESS="$MAIN_ORACLE_ADDRESS" FUNCTION_SIG="setTokenConfig(TokenConfig)" BROADCAST=true just admin-oracle-acm-give-call-permission

MODE=LIVE NETWORK=arbitrum ACCOUNT_TO_PERMIT="$GLOBALCONFIG_ADDRESS" \
RESILIENT_ORACLE_ADDRESS="$RESILIENT_ORACLE_ADDRESS" \
TARGET_ADDRESS="$MAIN_ORACLE_ADDRESS" \
FUNCTION_SIG="setTokenConfig((address,address,uint256))" \
BROADCAST=true just admin-oracle-acm-give-call-permission

MODE=LIVE NETWORK=arbitrum ACCOUNT_TO_PERMIT="$GLOBALCONFIG_ADDRESS" \
RESILIENT_ORACLE_ADDRESS="$RESILIENT_ORACLE_ADDRESS" \
TARGET_ADDRESS="$BOUND_VALIDATOR_ADDRESS" \
FUNCTION_SIG="setValidateConfig(ValidateConfig)" \
BROADCAST=true just admin-oracle-acm-give-call-permission

MODE=LIVE NETWORK=arbitrum ACCOUNT_TO_PERMIT="$GLOBALCONFIG_ADDRESS" \
RESILIENT_ORACLE_ADDRESS="$RESILIENT_ORACLE_ADDRESS" \
TARGET_ADDRESS="$RESILIENT_ORACLE_ADDRESS" \
FUNCTION_SIG="setTokenConfig(TokenConfig)" \
BROADCAST=true just admin-oracle-acm-give-call-permission
